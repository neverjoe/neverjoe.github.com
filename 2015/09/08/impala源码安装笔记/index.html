    • 本文是采用git clone源码，部署单机cluster impala集群。目的是debug和围观impala源码，部署生产环境的请绕行。

    环境

    • ubuntu14.04.1 64bit, 16G Memory, Intel(R) Core(TM) i7-2600 CPU @ 3.40GHz, 1TB disk
    • gcc/g++ 4.8.2

    准备相关依赖

    java

    安装java最新版即可。

    postgresql && boost

    配置postgresql

    postgresql认证

    编辑 /etc/postgresql/*/main/pg_hba.conf 文件。由于源码中默认user为hiveuser，包括metastore在内的user均为hiveuser，password均为‘password’。

    而

    sudo service postgresql restart

    sudo -u postgres psql postgres
    postgres=# CREATE ROLE hiveuser LOGIN PASSWORD ‘password’;
    postgres=# ALTER ROLE hiveuser WITH CREATEDB;

    cd /usr/lib/x86_64-linux-gnu
    sudo ln -s libboost_filesystem.so libboost_filesystem-mt.so
    sudo ln -s libboost_filesystem.a libboost_filesystem-mt.a
    sudo ln -s libboost_system.so libboost_system-mt.so
    sudo ln -s libboost_system.a libboost_system-mt.a
    sudo ln -s libboost_regex.so libboost_regex-mt.so
    sudo ln -s libboost_regex.a libboost_regex-mt.a

    wget http://llvm.org/releases/3.3/llvm-3.3.src.tar.gz
    tar xvf llvm-3.3.src.tar.gz
    cd llvm-3.3.src/tools/
    svn co http://llvm.org/svn/llvm-project/cfe/tags/RELEASE_33/final/ clang
    cd ../projects/
    svn co http://llvm.org/svn/llvm-project/compiler-rt/tags/RELEASE_33/final compiler-rt
    cd ..
    ./configure —with-pic
    make -j4 REQUIRES_RTTI=1
    sudo make install

    wget http://www.interior-dsgn.com/apache/maven/maven-3/3.0.5/binaries/apache-maven-3.0.5-bin.tar.gz
    tar xvf apache-maven-3.0.5-bin.tar.gz && sudo mv apache-maven-3.0.5 /usr/local

    export JAVA_HOME=/usr/lib/jvm/java-7-oracle
    export IMPALA_HOME=
    export BOOST_LIBRARYDIR=/usr/lib/x86_64-linux-gnu
    export LD_LIBRARY_PATH=/usr/lib/x86_64-linux-gnu
    export LC_ALL=”en_US.UTF-8”
    export M2_HOME=/usr/local/apache-maven-3.0.5
    export M2=$M2_HOME/bin
    export PATH=$M2:$PATH

    sudo mkdir /var/lib/hadoop-hdfs/
    sudo chown /var/lib/hadoop-hdfs/

    ssh-keygen -t dsa

    Do not type in any passkey. Just press enter.

    cat ~/.ssh/id_dsa.pub >> ~/.ssh/authorized_keys

    git clone https://github.com/cloudera/Impala.git

    cd ${IMPALA_HOME}
    source bin/impala-config.sh

    ${IMPALA_HOME}/buildall.sh -noclean -skiptests -build_shared_libs -format

    ${IMPALA_HOME}/buildall.sh -skiptests -build_shared_libs

    $.{IMPALA_HOME}/bin/make_debug.sh [-notests] -build_shared_libs

    cd ${IMPALA_HOME}/fe && mvn clean package dependency:copy-dependencies -DskipTests=true

    ${IMPALA_HOME}/testdata/bin/run-all.sh

    ${IMPALA_HOME}/bin/start-impala-cluster.py

    $cd ${IMPALA_HOME}
    $impala-shell.sh -q”SELECT version()”
    ```
